import { NextResponse } from "next/server";
import { db } from "@/lib/db";
import { CSCSCheck } from "@/lib/db/schema";

export async function POST(req: Request) {
  try {
    const { surname, nationalInsurance } = await req.json();

    // Fake verification result
    const result = {
      status: "not_found",
      message: "No CSCS record found for provided details.",
    };

    // Insert JSON as STRING so PostgreSQL accepts it
    const inserted = await db
      .insert(CSCSCheck)
      .values({
        surname,
        nationalInsurance,
        resultStatus: result.status,
        rawResponse: JSON.stringify(result),  // <-- FIXED (required)
      })
      .returning();

    return NextResponse.json(
      {
        ok: true,
        data: inserted[0],
      },
      { status: 200 }
    );
  } catch (error: any) {
    console.error("CSCS ERROR:", error);
    return NextResponse.json(
      {
        ok: false,
        error: error.message,
      },
      { status: 500 }
    );
  }
}
